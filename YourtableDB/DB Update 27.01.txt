CREATE SEQUENCE IF NOT EXISTS public._ID
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;
	
CREATE SEQUENCE IF NOT EXISTS public.QUESTION_ID
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;

CREATE TABLE _COMMENT (
_ID INT primary key NOT NULL DEFAULT NEXTVAL('public._ID'),
_RESTAURANT_ID INT,
CUSTOMER_ID INT,
_COMMENT text,
CONSTRAINT FK_C_ID FOREIGN key (CUSTOMER_ID) references CUSTOMER(CUSTOMER_ID),
CONSTRAINT FK_R_ID FOREIGN key (_RESTAURANT_ID) references RESTAURANT(ID)
);

CREATE TABLE QUESTION (
QUESTION_ID INT primary key NOT NULL DEFAULT NEXTVAL('public.QUESTION_ID'),
_RESTAURANT_ID INT,
CUSTOMER_ID INT,
QUESTION text,
CONSTRAINT FK_C_ID FOREIGN key (CUSTOMER_ID) references CUSTOMER(CUSTOMER_ID),
CONSTRAINT FK_R_ID FOREIGN key (_RESTAURANT_ID) references RESTAURANT(ID)
);
